83 Pybind11Helper {
84  using Element = typename ndarray::Array<T,N,C>::Element;
85  using Wrapper = pybind11::array_t<typename std::remove_const<Element>::type, 0>; // 0: no ensurecopy
86  static constexpr bool isConst = std::is_const<Element>::value;

88  Pybind11Helper() : isNone(false), wrapper() {}

90  bool init(pybind11::handle src) {
91  isNone = src.is_none();
92  if (isNone) {
93  return true;
94  }
95  if (!Wrapper::check_(src)) {
96  return false;
97  }
98  try {
99  wrapper = pybind11::reinterpret_borrow<Wrapper>(src);
100  } catch (pybind11::error_already_set & err) {
101  return false;
102  }
103  return true;
104  }
105 
106  bool check() const {
107  if (isNone) {
108  return true;
109  }
110  if (!wrapper) {
111  return false;
112  }
113  if (wrapper.ndim() != N) {
114  return false;
115  }
116  if (!isConst && !wrapper.writeable()) {
117  return false;
118  }
119  pybind11_np_size_t const * shape = wrapper.shape();
120  pybind11_np_size_t const * strides = wrapper.strides();
121  pybind11_np_size_t const itemsize = wrapper.itemsize();
122  if (C > 0) {
123  pybind11_np_size_t requiredStride = itemsize;
124  for (int i = 0; i < C; ++i) {
125  if (strides[N-i-1] != requiredStride) {
126  return false;
127  }
128  requiredStride *= shape[N-i-1];
129  }
130  } else if (C < 0) {
131  pybind11_np_size_t requiredStride = itemsize;
132  for (int i = 0; i < -C; ++i) {
133  if (strides[i] != requiredStride) {
134  return false;
135  }
136  requiredStride *= shape[i];
137  }
138  }
139  return true;
140  }

142  ndarray::Array<T,N,C> convert() const {
143  if (isNone) {
144  return ndarray::Array<T,N,C>();
145  }
146  if (!pybind11::reinterpret_borrow<pybind11::bool_>(wrapper.dtype().attr("isnative"))) {
147  PyErr_SetString(PyExc_TypeError, "Only arrays with native byteorder can be converted to C++.");
148  throw pybind11::error_already_set();
149  }
150  Vector<ndarray::Size,N> nShape;
151  Vector<ndarray::Offset,N> nStrides;
152  pybind11_np_size_t const * pShape = wrapper.shape();
153  pybind11_np_size_t const * pStrides = wrapper.strides();
154  pybind11_np_size_t const itemsize = wrapper.itemsize();
155  for (int i = 0; i < N; ++i) {
156  if (pStrides[i] % itemsize != 0) {
157  PyErr_SetString(
158  PyExc_TypeError,
159  "Cannot convert array to C++: strides must be an integer multiple of the element size"
160  );
161  throw pybind11::error_already_set();
162  }
163  nShape[i] = pShape[i];
164  nStrides[i] = pStrides[i]/itemsize;
165  }
166  return ndarray::Array<T,N,C>(
167  ndarray::external(const_cast<Element*>(wrapper.data()),
168  nShape, nStrides, pybind11::object(wrapper))
169  );
170  }

172  static pybind11::handle toPython(ndarray::Array<T,N,C> const & src) {
173  Vector<Size,N> nShape = src.getShape();
174  Vector<Offset,N> nStrides = src.getStrides();
175  std::vector<pybind11_np_size_t> pShape(N);
176  std::vector<pybind11_np_size_t> pStrides(N);
177  pybind11_np_size_t const itemsize = sizeof(Element);
178  for (int i = 0; i < N; ++i) {
179  pShape[i] = nShape[i];
180  pStrides[i] = nStrides[i]*itemsize;
181  }
182  pybind11::object base;
183  if (src.getManager()) {
184  base = pybind11::reinterpret_steal<pybind11::object>(ndarray::makePyManager(src.getManager()));
185  }
186  Wrapper result(pShape, pStrides, src.getData(), base);
187  if (std::is_const<Element>::value) {
188  result.attr("flags")["WRITEABLE"] = false;
189  }
190  return result.release();
191  }

194  Wrapper wrapper;
195 };
